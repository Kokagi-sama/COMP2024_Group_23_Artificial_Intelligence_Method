import java.util.ArrayList;
import java.util.Collections;
import java.util.List;
import java.util.Random;

public class EvolutionaryFirstFit {
    private List<BinPackingProblem> problems;
    private Random random = new Random();
    private int populationSize = 50;
    private int generations = 100;
    private double mutationRate = 0.05;

    public EvolutionaryFirstFit(List<BinPackingProblem> problems) {
        this.problems = problems;
    }

    public void solve() {
        for (BinPackingProblem problem : problems) {
            List<List<Bin>> population = initializePopulation(problem, populationSize);
            for (int i = 0; i < generations; i++) {
                List<List<Bin>> newPopulation = new ArrayList<>();

                // Selection
                Collections.sort(population, (o1, o2) -> Integer.compare(o1.size(), o2.size())); // Ascending order of bins used
                newPopulation.addAll(population.subList(0, populationSize / 2)); // Take the best half

                // Crossover and Mutation
                while (newPopulation.size() < populationSize) {
                    List<Bin> parent1 = selectRandomly(population);
                    List<Bin> parent2 = selectRandomly(population);
                    List<Bin> child = crossover(parent1, parent2);
                    if (random.nextDouble() < mutationRate) {
                        mutate(child);
                    }
                    newPopulation.add(child);
                }

                population = newPopulation;
            }

            // The best solution
            List<Bin> bestSolution = population.get(0);
            displaySolution(bestSolution);
        }
    }

    private List<Bin> crossover(List<Bin> parent1, List<Bin> parent2) {
        // Simplified crossover logic: take some bins from parent1 and some from parent2
        int crossoverPoint = random.nextInt(Math.min(parent1.size(), parent2.size()));
        List<Bin> child = new ArrayList<>();
        child.addAll(parent1.subList(0, crossoverPoint));
        child.addAll(parent2.subList(crossoverPoint, parent2.size()));
        return child;
    }

    private void mutate(List<Bin> solution) {
        // Simplified mutation logic: randomly select an item and move it to a new or existing bin
        if (solution.isEmpty()) return;
        int binIndex = random.nextInt(solution.size());
        Bin bin = solution.get(binIndex);
        if (!bin.items.isEmpty()) {
            Item item = bin.items.remove(random.nextInt(bin.items.size()));
            bin.remainingCapacity += item.getWeight();

            binIndex = random.nextInt(solution.size() + 1); // Maybe create a new bin
            if (binIndex == solution.size()) {
                Bin newBin = new Bin(bin.remainingCapacity);
                newBin.items.add(item);
                newBin.remainingCapacity -= item.getWeight();
                solution.add(newBin);
            } else {
                solution.get(binIndex).items.add(item);
            }
        }
    }

    private List<List<Bin>> initializePopulation(BinPackingProblem problem, int populationSize) {
        List<List<Bin>> population = new ArrayList<>();
        for (int i = 0; i < populationSize; i++) {
            population.add(firstFit(problem));
        }
        return population;
    }

    private List<Bin> selectRandomly(List<List<Bin>> population) {
        return population.get(random.nextInt(population.size()));
    }

    private void displaySolution(List<Bin> solution) {
        System.out.println("Best solution uses " + solution.size() + " bins:");
        for (Bin bin : solution) {
            System.out.println("- Bin with remaining capacity " + bin.remainingCapacity + ":");
            for (Item item : bin.items) {
                System.out.println("  Item weight: " + item.getWeight());
            }
        }
    }

            this.items = new ArrayList<>();
            this.remainingCapacity = capacity;
        }
    }

        int weight;

        public Item(int weight) {
            this.weight = weight;
        }

        public int getWeight() {
            return weight;
